### Checkpoint B — Script audit acomptes overflow

**Contexte** :
V46 a fixé le bug de validation acomptes pour les NOUVEAUX devis.
Mais les devis EXISTANTS peuvent avoir sum(acomptes) > montant_total.
Il faut auditer la BD pour identifier les incohérences.

**Fichier à créer** : `scripts/audit-acomptes-overflow.js`

**Logique** :
```javascript
// scripts/audit-acomptes-overflow.js
const admin = require('firebase-admin');
const sa = require('../serviceAccountKey.json');
admin.initializeApp({ credential: admin.credential.cert(sa) });
const db = admin.firestore();

(async () => {
  console.log('═══ AUDIT ACOMPTES OVERFLOW ═══\n');

  // 1. Récupérer tous les devis
  const devisSnap = await db.collection('quotes').get();
  console.log(`Total devis : ${devisSnap.size}`);

  const incoherences = [];

  for (const devisDoc of devisSnap.docs) {
    const devis = devisDoc.data();
    const devisId = devisDoc.id;
    const montantTotal = devis.montant_total || 0;

    // 2. Récupérer tous les acomptes liés
    const acomptesSnap = await db.collection('acomptes')
      .where('devis_id', '==', devisId)
      .get();

    if (acomptesSnap.empty) continue;

    // 3. Calculer le total
    let totalAcomptes = 0;
    const detailsAcomptes = [];
    acomptesSnap.forEach(doc => {
      const a = doc.data();
      totalAcomptes += a.montant || 0;
      detailsAcomptes.push({
        id: doc.id,
        montant: a.montant,
        statut: a.statut,
        date: a.date_creation,
      });
    });

    // 4. Vérifier overflow
    if (totalAcomptes > montantTotal) {
      incoherences.push({
        devis_id: devisId,
        client: devis.client_nom || 'N/A',
        montant_total: montantTotal,
        total_acomptes: totalAcomptes,
        overflow: totalAcomptes - montantTotal,
        nb_acomptes: detailsAcomptes.length,
        details: detailsAcomptes,
      });
    }

    // 5. Vérifier nb acomptes > 3 (règle business)
    if (detailsAcomptes.length > 3) {
      console.warn(`⚠️  ${devisId} : ${detailsAcomptes.length} acomptes (max 3)`);
    }
  }

  // 6. Rapport
  console.log(`\n═══ RÉSULTATS ═══`);
  console.log(`Devis avec overflow : ${incoherences.length}`);

  if (incoherences.length > 0) {
    console.log('\nDétails :');
    incoherences.forEach(inc => {
      console.log(`\n📌 ${inc.devis_id} (${inc.client})`);
      console.log(`   Montant total : ${inc.montant_total} €`);
      console.log(`   Total acomptes: ${inc.total_acomptes} €`);
      console.log(`   OVERFLOW     : +${inc.overflow.toFixed(2)} €`);
      console.log(`   Nb acomptes  : ${inc.nb_acomptes}`);
    });

    // 7. Export CSV
    const csv = ['devis_id;client;montant_total;total_acomptes;overflow;nb_acomptes'];
    incoherences.forEach(inc => {
      csv.push([
        inc.devis_id,
        inc.client,
        inc.montant_total,
        inc.total_acomptes,
        inc.overflow.toFixed(2),
        inc.nb_acomptes,
      ].join(';'));
    });

    const fs = require('fs');
    const filename = `audit-acomptes-overflow-${new Date().toISOString().slice(0,10)}.csv`;
    fs.writeFileSync(`./audits/${filename}`, csv.join('\n'));
    console.log(`\n✅ Rapport CSV : ./audits/${filename}`);
  } else {
    console.log('✅ Aucun overflow détecté.');
  }

  process.exit(0);
})();
```

### Checkpoint D — Tests automatisés pour validation acomptes

**Objectif** : Empêcher régression future du fix V46 acomptes.

**Approche simple** : tests Vitest (déjà dans le projet ?)

```typescript
// tests/acomptes-validation.test.ts
import { describe, it, expect } from 'vitest';
import { validerNouvelAcompte } from '../src/lib/acomptesHelpers';

describe('Validation acomptes V46', () => {
  it('refuse acompte qui dépasse le solde restant', () => {
    expect(() => validerNouvelAcompte({
      acomptes_existants: [{ montant: 20000 }],
      montant_total_devis: 25000,
      nouvel_acompte: 6000, // total = 26000 > 25000
    })).toThrow(/Acompte impossible/);
  });

  it('autorise acompte = solde restant', () => {
    expect(() => validerNouvelAcompte({
      acomptes_existants: [{ montant: 20000 }],
      montant_total_devis: 25000,
      nouvel_acompte: 5000,
    })).not.toThrow();
  });

  it('refuse 4ème acompte', () => {
    expect(() => validerNouvelAcompte({
      acomptes_existants: [
        { montant: 5000 }, { montant: 5000 }, { montant: 5000 }
      ],
      montant_total_devis: 25000,
      nouvel_acompte: 1000,
    })).toThrow(/maximum 3 acomptes/);
  });
});
```
