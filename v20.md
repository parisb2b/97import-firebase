```javascript
// scripts/test-functions.js
// Tests automatiques de toutes les fonctions critiques

const tests = {
  // 1. NUMÉROTATION DOCUMENTS
  'getNextNumber-DVS': async () => {
    const num = await getNextNumber('DVS');
    assert(num.startsWith('DVS-'), `Expected DVS-XXXX, got ${num}`);
    assert(num.length >= 10, `Number too short: ${num}`);
    return { status: 'PASS', value: num };
  },

  // 2. CALCUL PRIX PAR RÔLE
  'getDisplayPrice-user': () => {
    const price = getDisplayPrice({ prix_achat_cny: 10000 }, 'user', 7.82);
    const expected = (10000 / 7.82) * 2;
    assert(Math.abs(price - expected) < 0.01, `Expected ${expected}, got ${price}`);
    return { status: 'PASS', value: price };
  },

  // 3. SCORE COMPLÉTUDE PRODUIT
  'scoreCompletude-empty': () => {
    const score = scoreCompletude({});
    assert(score === 0, `Expected 0, got ${score}`);
    return { status: 'PASS', value: score };
  },

  'scoreCompletude-full': () => {
    const product = {
      categorie: 'mini-pelles', prix_achat_cny: 30000,
      nom_fr: 'Test', nom_zh: 'Test', nom_en: 'Test',
      description_fr: 'Test', code_hs: '8429.52',
      poids_net_kg: 2200, photos: ['url1']
    };
    const score = scoreCompletude(product);
    assert(score === 100, `Expected 100, got ${score}`);
    return { status: 'PASS', value: score };
  },

  // 4. FORMATAGE DATES
  'formatDate-paris': () => {
    const date = new Date('2026-04-11T10:00:00Z');
    const formatted = formatDateParis(date);
    assert(formatted.includes('2026'), `Date format error: ${formatted}`);
    return { status: 'PASS', value: formatted };
  },

  // 5. GÉNÉRATION PDF (smoke test)
  'generateDevis-smoke': async () => {
    const mockQuote = {
      numero: 'DVS-TEST-001',
      client_nom: 'Test Client',
      client_adresse: '1 Rue Test, 97200 Fort-de-France',
      lignes: [{ ref: 'TEST-001', nom_fr: 'Test', qte: 1, prix_unitaire: 1000, total: 1000 }],
      total_ht: 1000, acompte_pct: 30
    };
    try {
      const pdf = await generateDevis(mockQuote);
      assert(pdf !== null, 'PDF generation returned null');
      return { status: 'PASS', value: 'PDF generated OK' };
    } catch (e) {
      return { status: 'FAIL', error: e.message };
    }
  },
};

// Exécuter tous les tests
async function runAllTests() {
  const results = [];
  for (const [name, test] of Object.entries(tests)) {
    try {
      const result = await test();
      results.push({ name, ...result });
      console.log(`${result.status === 'PASS' ? '✅' : '❌'} ${name}: ${result.value || result.error}`);
    } catch (e) {
      results.push({ name, status: 'ERROR', error: e.message });
      console.log(`💥 ${name}: ${e.message}`);
    }
  }
  
  const passed = results.filter(r => r.status === 'PASS').length;
  const failed = results.filter(r => r.status !== 'PASS').length;
  console.log(`\n📊 RÉSULTATS: ${passed}/${results.length} tests passés, ${failed} échecs`);
  return results;
}

runAllTests();
```
